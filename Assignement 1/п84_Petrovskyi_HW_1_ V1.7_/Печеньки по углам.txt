import UIKit
//  All robot commands can be founded in GameViewController.h
class SwiftRobotControlCenter: RobotControlCenter {
    
    //  Level name setup
    override func viewDidLoad() {
        levelName = "L4H" //  Level name
        super.viewDidLoad()
    }
    
    override func run() {
        markCorners()
    }
    
    /// Mark all corners main func
    func markCorners() {
        while noCandyPresent {
            findTheWay()
            findCorner()
            nextCandy()
        }
    }
    
    /// Next candy
    func nextCandy() {
        nextCorner()
        put()
    }
    
    /// Find The way
    func findTheWay() {
        if frontIsClear {
            turnLeft()
            moveNearestWall()
        } else if rightIsClear {
            turnRight()
            moveNearestWall()
        } else if leftIsClear {
            moveNearestWall()
        } else{
            backTurn()
            moveNearestWall()
        }
    }
    
    /// Find the corner
    func findCorner() {
        findTheWay()
        put()
    }
    
    /// Go nearest wall
    func moveNearestWall() {
        while frontIsClear {
            move()
        }
    }
    
    /// Next corner
    func nextCorner() {
        findTheWay()
        while frontIsClear {
            move()
        }
        put()
        findTheWay()
        if noCandyPresent {
            put()
            findTheWay()
        }
    }
    
    /// Turn facing left
    func turnLeft() {
        for _ in 1...3 {
            turnRight()
        }
    }
    
    /// Back turn
    func backTurn() {
        for _ in 1...2 {
            turnLeft()
        }
    }
    
}
